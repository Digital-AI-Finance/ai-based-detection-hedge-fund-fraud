% ==================== SLIDE 4: Boolean Query Structure ====================
\begin{frame}{Boolean Query Structure}

\begin{block}{Query Design}
Primary search query combines three conceptual blocks using Boolean operators to capture interdisciplinary literature.
\end{block}

\vspace{2mm}

\begin{columns}[T]

\column{0.32\textwidth}
\begin{block}{Block 1: Financial Domain}
\compactlist
\begin{itemize}
    \item ``hedge fund''
    \item ``investment fund''
    \item ``alternative investment''
\end{itemize}
\vspace{1mm}
\textcolor{mlpurple}{\textbf{OR}} operator
\end{block}

\column{0.32\textwidth}
\begin{block}{Block 2: Fraud Terms}
\compactlist
\begin{itemize}
    \item ``fraud''
    \item ``manipulation''
    \item ``anomaly''
\end{itemize}
\vspace{1mm}
\textcolor{mlpurple}{\textbf{OR}} operator
\end{block}

\column{0.32\textwidth}
\begin{block}{Block 3: AI/ML Methods}
\compactlist
\begin{itemize}
    \item ``machine learning''
    \item ``artificial intelligence''
    \item ``deep learning''
    \item ``neural network''
\end{itemize}
\textcolor{mlpurple}{\textbf{OR}} operator
\end{block}

\end{columns}

\vspace{3mm}

\begin{center}
\fcolorbox{mlpurple}{mllavender4}{%
\begin{minipage}{0.92\textwidth}
\centering
\textbf{Primary Query:} \\[1mm]
(Block 1) \textcolor{mlpurple}{\textbf{AND}} (Block 2) \textcolor{mlpurple}{\textbf{AND}} (Block 3)
\end{minipage}
}
\end{center}

\vspace{2mm}

\textbf{Secondary Query:} Broader terms to capture related research
\begin{itemize}\compactlist
    \item (``fund'' OR ``portfolio'') AND (``fraud detection'' OR ``anomaly detection'') AND (``classification'' OR ``prediction'')
    \item Captures mutual fund and general portfolio fraud studies
\end{itemize}

\bottomnote{Boolean query designed to balance precision and recall across interdisciplinary literature.}
\end{frame}

% ==================== SLIDE 6: Inclusion and Exclusion Criteria ====================
\begin{frame}{Inclusion and Exclusion Criteria}

\begin{columns}[T]

\column{0.48\textwidth}
\begin{block}{Inclusion Criteria}
\compactlist
\begin{enumerate}
    \item \textbf{Domain:} Fraud, manipulation, or anomaly detection in investment funds or alternative investments

    \item \textbf{Methods:} Employed or substantively discussed AI/ML methods for detection, prediction, or classification

    \item \textbf{Quality:} Peer-reviewed publications or widely cited preprints (minimum 20 citations)

    \item \textbf{Language:} Published in English
\end{enumerate}
\end{block}

\column{0.48\textwidth}
\begin{block}{Exclusion Criteria}
\compactlist
\begin{enumerate}
    \item \textbf{Pure Methodology:} Papers without financial application or validation

    \item \textbf{Different Domain:} Studies focused exclusively on credit card, payment, or transaction-level fraud without fund-level components

    \item \textbf{Duplicates:} Multiple publications reporting the same study across venues
\end{enumerate}
\end{block}

\end{columns}

\vspace{4mm}

\begin{center}
\fcolorbox{mlpurple}{mllavender3}{%
\begin{minipage}{0.9\textwidth}
\centering
\textbf{Quality Threshold:} Peer-reviewed publications or preprints with at least 20 citations ensures rigor and impact while capturing emerging research.
\end{minipage}
}
\end{center}

\bottomnote{Criteria designed to ensure relevance, quality, and comprehensive coverage.}
\end{frame}

% ==================== SLIDE 7: Screening Process ====================
\begin{frame}{Screening Process and Results}

\begin{block}{Multi-Stage Screening}
Two independent reviewers conducted screening with disagreements resolved through discussion and third-reviewer consultation.
\end{block}

\vspace{2mm}

\begin{columns}[T]

\column{0.48\textwidth}
\textbf{Screening Stages:}

\begin{enumerate}\compactlist
    \item \textbf{Database Search:} Initial queries across 5 databases
    \vspace{1mm}

    \item \textbf{Title/Abstract Screening:} Relevance assessment
    \vspace{1mm}

    \item \textbf{Full-Text Review:} Detailed examination against inclusion criteria
    \vspace{1mm}

    \item \textbf{Snowballing:} Manual review of highly cited paper references
\end{enumerate}

\vspace{3mm}

\textbf{Quality Assurance:}
\begin{itemize}\compactlist
    \item Two independent reviewers
    \item Disagreement resolution protocol
    \item Third reviewer consultation as needed
\end{itemize}
