\vspace{2mm}

\begin{itemize}\compactlist

\begin{frame}{Search Funnel: 500 $\to$ 120 $\to$ \systematicpapers{}}
\begin{columns}[T]
\column{0.55\textwidth}
\begin{enumerate}\compactlist
\item \textbf{Initial database searches}: $\sim$\textbf{500} potentially relevant publications
\vspace{3mm}
\item \textbf{Title and abstract screening}: reduced to \textbf{120} papers warranting full-text review
\vspace{3mm}
\item \textbf{Full-text review} against inclusion criteria:
  \begin{itemize}\compactlist
  \item Peer-reviewed or widely cited preprints
  \item Addresses fraud/anomaly detection in investment funds
  \item Uses AI/ML methods
  \end{itemize}
  $\Rightarrow$ \textbf{80 papers} met all criteria (core corpus)
\vspace{3mm}
\item \textbf{Contextual background}: +25 papers on financial regulation, general fraud detection, foundational ML
\vspace{3mm}
\item \textbf{Total systematically identified}: \textbf{\systematicpapers{} papers}
\end{enumerate}

\column{0.42\textwidth}

The synthesis focuses on \textbf{methodological insights and contextual applicability} rather than exhaustive enumeration.

\begin{frame}{Limitations: Proprietary Data and Licensing Constraints}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Proprietary Enforcement Datasets}
\begin{itemize}\compactlist
\item Several cited studies use \textbf{proprietary enforcement datasets} maintained by regulatory agencies
\item Cannot be independently verified due to confidentiality restrictions
\item We rely on published methodological descriptions
\item Noted where data access limitations prevent full reproducibility
\end{itemize}

\column{0.48\textwidth}
\textbf{Alternative Data Licensing}
\begin{itemize}\compactlist
\item Some referenced alternative data sources:
  \begin{itemize}\compactlist
  \item Satellite imagery
  \item Geolocation analytics
  \item Sentiment data
  \end{itemize}
\item Available only through \textbf{vendor-specific licenses}
\item Cannot be redistributed
\item We prioritize discussion of methods evaluable on \textbf{publicly accessible data}
\end{itemize}
\end{columns}
\vspace{3mm}
\begin{block}{Implication}
Full end-to-end reproducibility of hedge fund fraud detection research remains constrained by \textbf{data access limitations endemic to the domain}.
\end{block}
\bottomnote{Source: Paper Reproducibility Statement}
\end{frame}

% SLIDE 10 -- OP1 as Critical Enabler

\begin{frame}{OP1: The Critical Enabler for Reproducibility}
\begin{block}{The Fundamental Bottleneck}
Until public benchmark datasets exist (OP1), full end-to-end reproducibility of hedge fund fraud detection research will remain \textbf{constrained by data access limitations endemic to the domain}.
\end{block}
\vspace{3mm}
\begin{columns}[T]
\column{0.48\textwidth}
\textbf{What OP1 Would Unlock}
\begin{itemize}\compactlist
\item \textbf{Reproducible comparisons} across studies
\item Standardized evaluation protocols
\item Cross-method benchmarking on common data
\item Progress measurement for the field
\item \textbf{Accelerated methodological advancement} (as demonstrated by benchmarks in credit card fraud, image classification, NLP)
\end{itemize}

\column{0.48\textwidth}
\textbf{Two-Track Path Forward}
\begin{enumerate}\compactlist
\item \textcolor{mlblue}{\textbf{Synthetic benchmarks}}: regime-switching models with injected fraud patterns, calibrated to real distributions
\item \textcolor{mlblue}{\textbf{Anonymized regulatory data}}: differential privacy protocols enabling regulators to release sanitized enforcement-labeled datasets
\end{enumerate}
\vspace{3mm}
\textcolor{mlblue}{Researchers are encouraged to consult the full research agenda (Section 6) for detailed approaches.}
\end{columns}
\bottomnote{Source: Paper Reproducibility Statement; Section 6.1.1 (OP1)}
\end{frame}
